Be synthetic with PySynthetic¶

Example:

class ThisHurtsMyKeyboard:

    def __init__(self, a, b, c = None):
        """
    :type a: int
    :type b: str
    :type c: list|None
"""
        self._a = a
        self._b = b
        self._c = c

    @property
    def a(self):
        return self._a

    @a.setter
    def a(self, value):
        self._a = value

...
